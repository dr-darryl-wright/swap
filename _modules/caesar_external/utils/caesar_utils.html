    def reduce(cls, subject, data):
        """
        PUT subject score to Caesar
        """
        config = Config._config
        pan = cls.instance().pan

        endpoint = config.caesar_endpoint()
        path = config.workflow_path()

        body = {
            'reduction': {
                'subject_id': subject,
                'data': data
            }
        }

        r = pan.put(endpoint=endpoint, path=path, json=body)

        return r
